local autoCollectFruits = false -- Bi·∫øn b·∫≠t/t·∫Øt Auto Nh·∫∑t Tr√°i

-- H√†m d·ªãch chuy·ªÉn ƒë·∫øn tr√°i
local function collectFruit(fruit)
    if not fruit or not fruit:FindFirstChild("Handle") then return end
    local player = game.Players.LocalPlayer
    local char = player.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return end

    -- Bay ƒë·∫øn tr√°i (cao h∆°n m·ªôt ch√∫t ƒë·ªÉ tr√°nh l·ªói va ch·∫°m)
    char.HumanoidRootPart.CFrame = fruit.Handle.CFrame + Vector3.new(0, 3, 0)
end

-- H√†m t√¨m tr√°i g·∫ßn nh·∫•t v√† nh·∫∑t
local function autoCollect()
    while autoCollectFruits do
        local nearestFruit
        local minDist = math.huge
        local playerPos = game.Players.LocalPlayer.Character.HumanoidRootPart.Position

        -- T√¨m tr√°i g·∫ßn nh·∫•t
        for _, fruit in pairs(game.Workspace:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                local fruitPos = fruit.Handle.Position
                local dist = (playerPos - fruitPos).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestFruit = fruit
                end
            end
        end

        -- N·∫øu t√¨m th·∫•y tr√°i, di chuy·ªÉn ƒë·∫øn nh·∫∑t
        if nearestFruit then
            collectFruit(nearestFruit)
        end

        wait(1) -- Ki·ªÉm tra m·ªói gi√¢y
    end
end

-- H√†m b·∫≠t/t·∫Øt Auto Nh·∫∑t Tr√°i
local function toggleAutoCollect()
    autoCollectFruits = not autoCollectFruits

    if autoCollectFruits then
        autoCollect()
    end
end

AddButton(Tab6o, {
    Name = "Auto Nh·∫∑t Tr√°i",
    Callback = function()

    end

local function TeleportToSea(seaNumber)
    local npcName = ""

    if seaNumber == 2 then
        npcName = "Experienced Captain" -- NPC ·ªü SEA1 ƒë·ªÉ ƒëi SEA2
    elseif seaNumber == 3 then
        npcName = "King Red Head" -- NPC ·ªü SEA2 ƒë·ªÉ ƒëi SEA3
    elseif seaNumber == 1 then
        npcName = "Elite Hunter" -- NPC ·ªü SEA3 ƒë·ªÉ v·ªÅ SEA1
    end

    -- T√¨m NPC chuy·ªÉn SEA
    for _, npc in pairs(game.Workspace.NPCs:GetChildren()) do
        if npc:IsA("Model") and npc.Name == npcName and npc:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 0, 3) -- ƒê·ª©ng g·∫ßn NPC
            task.wait(1)

            -- Gi·∫£ l·∫≠p nh·∫•n chu·ªôt ƒë·ªÉ t∆∞∆°ng t√°c NPC
            vim:SendMouseButtonEvent(0, 0, 0, true, game, 1)
            task.wait(0.1)
            vim:SendMouseButtonEvent(0, 0, 0, false, game, 1)

            -- Ch·ªù m·ªü menu SEA v√† x√°c nh·∫≠n chuy·ªÉn SEA
            task.wait(2)
            vim:SendKeyEvent(true, Enum.KeyCode.Return, false, game)
            task.wait(0.1)
            vim:SendKeyEvent(false, Enum.KeyCode.Return, false, game)
            
            break
        end
    end
end

local espEnabled = false -- Tr·∫°ng th√°i b·∫≠t/t·∫Øt ESP

local function createESP(player)
    if player == game.Players.LocalPlayer then return end -- Kh√¥ng hi·ªÉn th·ªã b·∫£n th√¢n

    local character = player.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then return end

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = character.HumanoidRootPart
    billboard.Size = UDim2.new(0, 200, 0, 50) -- K√≠ch th∆∞·ªõc ban ƒë·∫ßu
    billboard.StudsOffset = Vector3.new(0, 3, 0) -- ƒê·∫∑t ·ªü gi·ªØa nh√¢n v·∫≠t
    billboard.AlwaysOnTop = true

    local textLabel = Instance.new("TextLabel", billboard)
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3 = Color3.new(1, 0, 0) -- M√†u ƒë·ªè
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.TextStrokeTransparency = 0.2 -- Vi·ªÅn ch·ªØ gi√∫p d·ªÖ nh√¨n h∆°n

    billboard.Parent = character.HumanoidRootPart

    -- C·∫≠p nh·∫≠t v·ªã tr√≠ & k√≠ch th∆∞·ªõc theo kho·∫£ng c√°ch
    spawn(function()
        while espEnabled and character and character:FindFirstChild("Humanoid") and character:FindFirstChild("HumanoidRootPart") do
            local humanoid = character:FindFirstChild("Humanoid")
            local localPlayer = game.Players.LocalPlayer
            if localPlayer and localPlayer.Character and localPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local distance = (localPlayer.Character.HumanoidRootPart.Position - character.HumanoidRootPart.Position).Magnitude
                
                -- K√≠ch th∆∞·ªõc ESP thay ƒë·ªïi theo kho·∫£ng c√°ch
                local scale = math.clamp(distance / 10, 1, 10) -- Gi·ªõi h·∫°n t·ªâ l·ªá t·ª´ 1 ƒë·∫øn 10 l·∫ßn
                local size = 20 * scale -- Khi ·ªü g·∫ßn, k√≠ch th∆∞·ªõc l√† 20, khi xa tƒÉng d·∫ßn

                billboard.Size = UDim2.new(0, size * 10, 0, size * 2) -- ESP l·ªõn h∆°n khi xa

                -- Hi·ªÉn th·ªã th√¥ng tin m√°u & kho·∫£ng c√°ch
                textLabel.Text = string.format("[%s] HP: %d | %dm", player.Name, math.floor(humanoid.Health), math.floor(distance))
            end
            wait(0.1)
        end
        billboard:Destroy()
    end)
end

-- H√†m b·∫≠t/t·∫Øt ESP
local function toggleESP()
    espEnabled = not espEnabled

    if espEnabled then
        -- Th√™m ESP cho t·∫•t c·∫£ ng∆∞·ªùi ch∆°i
        for _, player in pairs(game.Players:GetPlayers()) do
            if player ~= game.Players.LocalPlayer then
                createESP(player)
            end
        end

        -- L·∫Øng nghe ng∆∞·ªùi ch∆°i m·ªõi tham gia
        game.Players.PlayerAdded:Connect(function(player)
            if espEnabled then createESP(player) end
        end)
    else
        -- X√≥a to√†n b·ªô ESP khi t·∫Øt
        for _, player in pairs(game.Players:GetPlayers()) do
            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                for _, gui in pairs(player.Character.HumanoidRootPart:GetChildren()) do
                    if gui:IsA("BillboardGui") then gui:Destroy() end
                end
            end
        end
    end
end

local fruitESPEnabled = false -- Bi·∫øn b·∫≠t/t·∫Øt ESP Tr√°i

local function createFruitESP(fruit)

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = fruit.Handle
    billboard.Size = UDim2.new(3, 0, 1.5, 0)
    billboard.StudsOffset = Vector3.new(0, 2, 0)
    billboard.AlwaysOnTop = true

    local textLabel = Instance.new("TextLabel", billboard)
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3 = Color3.new(1, 1, 0) -- M√†u v√†ng
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.SourceSansBold

    billboard.Parent = fruit.Handle

    -- C·∫≠p nh·∫≠t th√¥ng tin kho·∫£ng c√°ch real-time
    spawn(function()
        while fruitESPEnabled and fruit and fruit:FindFirstChild("Handle") do

            local fruitPos = fruit.Handle.Position
            local distance = (playerPos - fruitPos).Magnitude
            textLabel.Text = string.format("üçè %s | %dm", fruit.Name, math.floor(distance))
            wait(0.2)
        end
        billboard:Destroy()
    end)
end

end
